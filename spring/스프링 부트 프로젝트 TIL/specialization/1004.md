```sql
UPDATE t_recipe_ingredient
SET ing_price  =ing_price /1000
WHERE rec_ing_seq in (SELECT t_ing.ris FROM
 (SELECT rec_ing_seq ris FROM t_recipe_ingredient_test WHERE ing_price_unit='L') t_ing);
```
